#SelectKBest & Chi2


from sklearn.feature_selection import SelectKBest
from sklearn.feature_selection import chi2

X = data.iloc[:, 0:4]
y = data.iloc[:, -1]

bestfeatures = SelectKBest(score_func=chi2, k=4)
fit = bestfeatures.fit(X,y)

dfscores = pd.DataFrame(fit.scores_) #score tiap feature
dfcolumns = pd.DataFrame(X.columns) #nama tiap feature

featureScores = pd.concat([dfcolumns,dfscores],axis=1) #buat tabel

featureScores.columns = ['Feature','Score']  #set nama kolom tabel
print(featureScores.nlargest(4,'Score'))  #tampilkan tabel
