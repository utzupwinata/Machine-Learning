# Fuzzy C-Means Clustering

data = X
jumlah_cluster = 3
m = 2

cntr, u, u0, d, jm, p, fpc = fuzz.cluster.cmeans(data, jumlah_cluster, m, error=0.005, maxiter=1000)

# Hal yang didapat dari operasi Fuzzy C-Means:
# 1. cntr = pusat dari masing-masing cluster
# 2. u    = matriks partisi akhir
# 3. u0   = matriks partisi awal (tebakan/random)
# 4. d    = matriks euclidian distance akhir
# 5. jm   = fungsi objektif pada tiap iterasi
# 6. p    = jumlah iterasi yang dilakukan (stop saat mencapai maxiter atau batas toleransi)
# 7. fpc  = koefisien partisi fuzzy akhir
