# MENETAPKAN RULE (IMPLIKASI)


# 1. IF permintaan Turun AND persediaan Banyak, then produksi Berkurang
# 2. IF permintaan Turun AND persediaan Sedikit, then produksi Berkurang
# 3. IF permintaan Naik AND persediaan Banyak, then produksi Bertambah
# 4. IF permintaan Naik AND persediaan Sedikit, then produksi Bertambah

# Sugeno = Mamdani = fmin pada INFERENSI

#Rule 1 (demand low & storage high)
rule1 = np.fmin(permintaan_level_lo, gudang_level_hi)

#Rule 2 (demand low & storage low)
rule2 = np.fmin(permintaan_level_lo, gudang_level_lo)

#Rule 3 (demand high & storage high)
rule3 = np.fmin(permintaan_level_hi, gudang_level_hi)

#Rule 4 (demand high & storage low)
rule4 = np.fmin(permintaan_level_hi, gudang_level_lo)

print("\nRule 1 = ", rule1)
print("Rule 2 = ", rule2)
print("Rule 3 = ", rule3)
print("Rule 4 = ", rule4, '\n')

Rule 1 =  0.25
Rule 2 =  0.25
Rule 3 =  0.4
